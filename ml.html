1. Linear Regression (Gradient Descent)

import numpy as np

class LinearRegressionGD:
    def __init__(self, lr=0.01, epochs=1000):
        self.lr = lr
        self.epochs = epochs

    def fit(self, X, y):
        X = np.array(X)
        y = np.array(y).flatten()

        n, d = X.shape
        self.w = np.zeros(d)
        self.b = 0

        for _ in range(self.epochs):
            y_hat = X @ self.w + self.b
            error = y_hat - y

            self.w -= self.lr * (X.T @ error) / n
            self.b -= self.lr * np.mean(error)

    def predict(self, X):
        return X @ self.w + self.b

2. Logistic Regression

import numpy as np

def sigmoid(z):
    return 1 / (1 + np.exp(-z))

def train_logistic(X, y, lr=0.01, epochs=1000):
    w = np.zeros(X.shape[1])
    b = 0

    for _ in range(epochs):
        z = X @ w + b
        y_hat = sigmoid(z)

        error = y_hat - y

        w -= lr * (X.T @ error) / len(X)
        b -= lr * np.mean(error)

    return w, b

def predict(X, w, b):
    return sigmoid(X @ w + b)

4. KNN (From Scratch)

import numpy as np

def knn(X_train, y_train, X_test, k=3):
    preds = []

    for x in X_test:
        dist = np.sqrt(((X_train - x) ** 2).sum(axis=1))
        idx = np.argsort(dist)[:k]
        preds.append(np.mean(y_train[idx]))

    return np.array(preds)

7. KMeans (Scratch)

import numpy as np

def kmeans(X, k=3, iters=10):
    centroids = X[np.random.choice(len(X), k, replace=False)]

    for _ in range(iters):
        labels = np.array([
            np.argmin([np.linalg.norm(x - c) for c in centroids])
            for x in X
        ])

        centroids = np.array([
            X[labels == i].mean(axis=0)
            for i in range(k)
        ])

    return labels, centroids
